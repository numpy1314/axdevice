22/// A set of emulated device types that can be accessed by a specific address range type.
23pub struct AxEmuDevices<R: DeviceAddrRange> {
24    emu_devices: Vec<Arc<dyn BaseDeviceOps<R>>>,
25}
26
27impl<R: DeviceAddrRange + 'static> AxEmuDevices<R> {
28    /// Creates a new [`AxEmuDevices`] instance.
29    pub fn new() -> Self {
30        Self {
31            emu_devices: Vec::new(),
32        }
33    }
34
35    /// Adds a device to the set.
36    pub fn add_dev(&mut self, dev: Arc<dyn BaseDeviceOps<R>>) {
37        self.emu_devices.push(dev);
38    }
39
40    // pub fn remove_dev(&mut self, ...)
41    //
42    // `remove_dev` seems to need something like `downcast-rs` to make sense. As it's not likely to
43    // be able to have a proper predicate to remove a device from the list without knowing the
44    // concrete type of the device.
45
46    /// Find a device by address.
47    pub fn find_dev(&self, addr: R::Addr) -> Option<Arc<dyn BaseDeviceOps<R>>> {
48        self.emu_devices
49            .iter()
50            .find(|&dev| dev.address_range().contains(addr))
51            .cloned()
52    }
53
54    /// Iterates over the devices in the set.
55    pub fn iter(&self) -> impl Iterator<Item = &Arc<dyn BaseDeviceOps<R>>> {
56        self.emu_devices.iter()
57    }
58
59    /// Iterates over the devices in the set mutably.
60    pub fn iter_mut(&mut self) -> impl Iterator<Item = &mut Arc<dyn BaseDeviceOps<R>>> {
61        self.emu_devices.iter_mut()
62    }
63}

79#[inline]
80fn log_device_io(
81    addr_type: &'static str,
82    addr: impl core::fmt::LowerHex,
83    addr_range: impl core::fmt::LowerHex,
84    read: bool,
85    width: AccessWidth,
86) {
87    let rw = if read { "read" } else { "write" };
88    trace!(
89        "emu_device {}: {} {:#x} in range {:#x} with width {:?}",
90        rw, addr_type, addr, addr_range, width
91    )
92}
93
94#[inline]
95fn panic_device_not_found(
96    addr_type: &'static str,
97    addr: impl core::fmt::LowerHex,
98    read: bool,
99    width: AccessWidth,
100) -> ! {
101    let rw = if read { "read" } else { "write" };
102    error!(
103        "emu_device {} failed: device not found for {} {:#x} with width {:?}",
104        rw, addr_type, addr, width
105    );
106    panic!("emu_device not found");
107}

124    /// According the emu_configs to init every  specific device
125    fn init(this: &mut Self, emu_configs: &Vec<EmulatedDeviceConfig>) {
126        for config in emu_configs {
127            match config.emu_type {
128                EmulatedDeviceType::InterruptController => {
129                    #[cfg(target_arch = "aarch64")]
130                    {
131                        this.add_mmio_dev(Arc::new(Vgic::new()));
132                    }
133                    #[cfg(not(target_arch = "aarch64"))]
134                    {
135                        warn!(
136                            "emu type: {} is not supported on this platform",
137                            config.emu_type
138                        );
139                    }
140                }
141                EmulatedDeviceType::GPPTRedistributor => {
142                    #[cfg(target_arch = "aarch64")]
143                    {
144                        const GPPT_GICR_ARG_ERR_MSG: &'static str =
145                            "expect 3 args for gppt redistributor (cpu_num, stride, pcpu_id)";
146
147                        let cpu_num = config
148                            .cfg_list
149                            .first()
150                            .copied()
151                            .expect(GPPT_GICR_ARG_ERR_MSG);
152                        let stride = config
153                            .cfg_list
154                            .get(1)
155                            .copied()
156                            .expect(GPPT_GICR_ARG_ERR_MSG);
157                        let pcpu_id = config
158                            .cfg_list
159                            .get(2)
160                            .copied()
161                            .expect(GPPT_GICR_ARG_ERR_MSG);
162
163                        for i in 0..cpu_num {
164                            let addr = config.base_gpa + i * stride;
165                            let size = config.length;
166                            this.add_mmio_dev(Arc::new(arm_vgic::v3::vgicr::VGicR::new(
167                                addr.into(),
168                                Some(size),
169                                pcpu_id + i,
170                            )));
171
172                            info!(
173                                "GPPT Redistributor initialized for vCPU {} with base GPA {:#x} and length {:#x}",
174                                i, addr, size
175                            );
176                        }
177                    }
178                    #[cfg(not(target_arch = "aarch64"))]
179                    {
180                        warn!(
181                            "emu type: {} is not supported on this platform",
182                            config.emu_type
183                        );
184                    }
185                }
186                EmulatedDeviceType::GPPTDistributor => {
187                    #[cfg(target_arch = "aarch64")]
188                    {
189                        this.add_mmio_dev(Arc::new(arm_vgic::v3::vgicd::VGicD::new(
190                            config.base_gpa.into(),
191                            Some(config.length),
192                        )));
193
194                        info!(
195                            "GPPT Distributor initialized with base GPA {:#x} and length {:#x}",
196                            config.base_gpa, config.length
197                        );
198                    }
199                    #[cfg(not(target_arch = "aarch64"))]
200                    {
201                        warn!(
202                            "emu type: {} is not supported on this platform",
203                            config.emu_type
204                        );
205                    }
206                }
207                EmulatedDeviceType::GPPTITS => {
208                    #[cfg(target_arch = "aarch64")]
209                    {
210                        let host_gits_base = config
211                            .cfg_list
212                            .first()
213                            .copied()
214                            .map(PhysAddr::from_usize)
215                            .expect("expect 1 arg for gppt its (host_gits_base)");
216
217                        this.add_mmio_dev(Arc::new(arm_vgic::v3::gits::Gits::new(
218                            config.base_gpa.into(),
219                            Some(config.length),
220                            host_gits_base,
221                            false,
222                        )));
223
224                        info!(
225                            "GPPT ITS initialized with base GPA {:#x} and length {:#x}, host GITS base {:#x}",
226                            config.base_gpa, config.length, host_gits_base
227                        );
228                    }
229                    #[cfg(not(target_arch = "aarch64"))]
230                    {
231                        warn!(
232                            "emu type: {} is not supported on this platform",
233                            config.emu_type
234                        );
235                    }
236                }
237                EmulatedDeviceType::IVCChannel => {
238                    if this.ivc_channel.is_none() {
239                        // Initialize the IVC channel range allocator
240                        this.ivc_channel = Some(Mutex::new(RangeAllocator::new(Range {
241                            start: config.base_gpa,
242                            end: config.base_gpa + config.length,
243                        })));
244                        info!(
245                            "IVCChannel initialized with base GPA {:#x} and length {:#x}",
246                            config.base_gpa, config.length
247                        );
248                    } else {
249                        warn!("IVCChannel already initialized, ignoring additional config");
250                    }
251                }
252                _ => {
253                    warn!(
254                        "Emulated device {}'s type {:?} is not supported yet",
255                        config.name, config.emu_type
256                    );
257                }
258            }
259        }
260    }

262    /// Allocates an IVC (Inter-VM Communication) channel of the specified size.
263    pub fn alloc_ivc_channel(&self, size: usize) -> AxResult<GuestPhysAddr> {
264        if size == 0 {
265            return ax_err!(InvalidInput, "Size must be greater than 0");
266        }
267        if !is_aligned_4k(size) {
268            return ax_err!(InvalidInput, "Size must be aligned to 4K");
269        }
270
271        if let Some(allocator) = &self.ivc_channel {
272            allocator
273                .lock()
274                .allocate_range(size)
275                .map_err(|e| {
276                    warn!("Failed to allocate IVC channel range: {:x?}", e);
277                    axerrno::ax_err_type!(NoMemory, "IVC channel allocation failed")
278                })
279                .map(|range| {
280                    debug!("Allocated IVC channel range: {:x?}", range);
281                    GuestPhysAddr::from_usize(range.start)
282                })
283        } else {
284            ax_err!(InvalidInput, "IVC channel not exists")
285        }
286    }
287
288    /// Releases an IVC channel at the specified address and size.
289    pub fn release_ivc_channel(&self, addr: GuestPhysAddr, size: usize) -> AxResult {
290        if size == 0 {
291            return ax_err!(InvalidInput, "Size must be greater than 0");
292        }
293        if !is_aligned_4k(size) {
294            return ax_err!(InvalidInput, "Size must be aligned to 4K");
295        }
296
297        if let Some(allocator) = &self.ivc_channel {
298            allocator
299                .lock()
300                .free_range(addr.as_usize()..addr.as_usize() + size);
301            Ok(())
302        } else {
303            ax_err!(InvalidInput, "IVC channel not exists")
304        }
305    }

372    /// Handle the MMIO read by GuestPhysAddr and data width, return the value of the guest want to read
373    pub fn handle_mmio_read(&self, addr: GuestPhysAddr, width: AccessWidth) -> AxResult<usize> {
374        if let Some(emu_dev) = self.find_mmio_dev(addr) {
375            log_device_io("mmio", addr, emu_dev.address_range(), true, width);
376
377            return emu_dev.handle_read(addr, width);
378        }
379        panic_device_not_found("mmio", addr, true, width);
380    }
